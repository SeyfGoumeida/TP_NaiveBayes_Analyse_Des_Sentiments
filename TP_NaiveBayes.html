3 - insertion d'une nouvelle colonne 'Result' =¶
  • 'BAD' si Rating = 1-2,
  • 'NEUTRAL' si Rating = 3
  • 'GOOD' si Rating = 4-5
In [360]:
for index, row in df.iterrows():
    if row.Rating <= 2:
        df.loc[index, 'Resultat'] = 'Bad'
    elif row.Rating == 3:
        df.loc[index, 'Resultat'] = 'Neutral'
    else:
        df.loc[index, 'Resultat'] = 'Good'
